number = 100 
angka = 0 
while angka != number:     
  angka = int(input("masukkan angka inputan:"))     
  if angka > number:         
    print ("angka inputan lebih besar")     
  elif angka &lt; number:         
    print ("angka inputan lebih kecil")     
  else:         
    print ("angkanya sama")
